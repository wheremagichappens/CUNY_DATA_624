The function should take arguments y (the time series), alpha (the smoothing parameter and level. It should return the forecast of the next observation in the series. Does it give the same forecast as ses()?

my_ses <- function(y, alpha, l0)
  {
  y_hat <- l0
  for(index in 1:length(y))
    {
   y_hat <- alpha*y[index] + (1 - alpha)*y_hat 
    }
  print(paste0("Forecast result by My function: ",
      as.character(y_hat)
      ))
  }

my_ses2 <- function(pars = c(alpha, l0), y) #modification to return the sum of squares 
  {
  error <- 0
  SSE <- 0
  alpha <- pars[1]
  l0 <- pars[2]
  y_hat <- l0
  
  for(index in 1:length(y)) #Code from 7.2
    {
    error <- y[index] - y_hat
    SSE <- SSE + error^2
    
    y_hat <- alpha*y[index] + (1 - alpha)*y_hat 
    }
  
  return(SSE)
}
